const { StretchGateNative: NativeModule } = NativeModules;

      return JSON.stringify({ mock: true });
    }
    return NativeModule.presentAppPicker();

    if (!isNativeAvailable) {
      console.log('[StretchGate] Mock: saveSelection', selectionJson);
      return;
    }
    NativeModule.saveSelection(selectionJson);
  },

  // Apply OS-level restrictions to blocked apps

    if (!isNativeAvailable) {
      console.log('[StretchGate] Mock: applyRestrictions');
      return;
    }
    NativeModule.applyRestrictions();
  },

  // Lift restrictions for N minutes (unlock window)

useEffect(() => {
  // Handle cold start deep link
  Linking.getInitialURL().then((url) => {
    if (url) handleDeepLink(url);
  });

  // Handle warm start deep link
  const sub = Linking.addEventListener('url', ({ url }) => {
    handleDeepLink(url);
  });

  return () => sub.remove();
}, []);

const handleDeepLink = (url: string) => {
  if (!url) return;
  try {
    const parsed = new URL(url);
    if (parsed.pathname.includes('stretch/session')) {
      const targetApp = parsed.searchParams.get('targetApp');
      const stretchId = parsed.searchParams.get('stretchId');
      const params: Record<string, string> = {};
      if (targetApp) params.targetApp = targetApp;
      if (stretchId) params.stretchId = stretchId;
      router.push({ pathname: '/stretch/session', params });
    }
  } catch (e) {
    console.warn('[StretchGate] Deep link parse error:', e);
  }
};

## FIX 13 — Full notification service

Create the file: artifacts/mobile/services/notifications.ts

import * as Notifications from 'expo-notifications';
import { Platform } from 'react-native';

// Configure how notifications appear when app is foregrounded
Notifications.setNotificationHandler({
  handleNotification: async () => ({
    shouldShowAlert: true,
    shouldPlaySound: false,
    shouldSetBadge: false,
  }),
});

export type NotificationTime = { hour: number; minute: number };

const NOTIF_IDS = {
  DAILY_REMINDER: 'daily-reminder',
  STREAK_ALERT: 'streak-alert',
};

// Request permission (call this after user opts in)
export async function requestNotificationPermission(): Promise<boolean> {
  if (Platform.OS === 'web') return false;
  const { status: existing } = await Notifications.getPermissionsAsync();
  if (existing === 'granted') return true;
  const { status } = await Notifications.requestPermissionsAsync();
  return status === 'granted';
}

// Schedule daily reminder at the user's chosen time
export async function scheduleDailyReminder(
  time: NotificationTime,
  currentStreak: number
): Promise<void> {
  if (Platform.OS === 'web') return;

  // Cancel existing daily reminder first
  await Notifications.cancelScheduledNotificationAsync(
    NOTIF_IDS.DAILY_REMINDER
  ).catch(() => {});

  const streakMsg = currentStreak > 0
    ? `🔥 ${currentStreak} day streak — keep it going`
    : "Time to move. Your body will thank you.";

  await Notifications.scheduleNotificationAsync({
    identifier: NOTIF_IDS.DAILY_REMINDER,
    content: {
      title: 'StretchGate',
      body: streakMsg,
      sound: false,
      data: { type: 'daily_reminder' },
    },
    trigger: {
      type: Notifications.SchedulableTriggerInputTypes.DAILY,
      hour: time.hour,
      minute: time.minute,
    },
  });
}

// Schedule a streak protection alert if user hasn't stretched by 8pm
export async function scheduleStreakAlert(
  currentStreak: number
): Promise<void> {
  if (Platform.OS === 'web' || currentStreak === 0) return;

  await Notifications.cancelScheduledNotificationAsync(
    NOTIF_IDS.STREAK_ALERT
  ).catch(() => {});

  await Notifications.scheduleNotificationAsync({
    identifier: NOTIF_IDS.STREAK_ALERT,
    content: {
      title: `${currentStreak} day streak at risk 🔥`,
      body: "Do one stretch before midnight to keep it alive.",
      sound: false,
      data: { type: 'streak_alert' },
    },
    trigger: {
      type: Notifications.SchedulableTriggerInputTypes.DAILY,
      hour: 20,
      minute: 0,
    },
  });
}

// Cancel all scheduled notifications (call when user disables reminders)
export async function cancelAllNotifications(): Promise<void> {
  if (Platform.OS === 'web') return;
  await Notifications.cancelAllScheduledNotificationsAsync();
}

// Call this when a session is completed to reschedule streak alert
// for next day (since they've already stretched today, push to tomorrow)
export async function handleSessionCompleted(
  currentStreak: number
): Promise<void> {
  if (Platform.OS === 'web') return;
  // Cancel today's streak alert since they completed a stretch
  await Notifications.cancelScheduledNotificationAsync(
    NOTIF_IDS.STREAK_ALERT
  ).catch(() => {});
  // Reschedule for tomorrow
  await scheduleStreakAlert(currentStreak + 1);
}
