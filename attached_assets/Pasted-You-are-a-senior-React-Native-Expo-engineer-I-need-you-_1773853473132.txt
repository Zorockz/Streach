## FIX 6 — stretches.ts: Add recent-stretch exclusion and duration filter to getRandomStretch

FILE: artifacts/mobile/constants/stretches.ts

PROBLEM: getRandomStretch picks randomly with no exclusion, so the same stretch repeats. Also ignores preferred duration.

FIX: Replace the getRandomStretch function with:

export function getRandomStretch(
  areas: BodyArea[] = [],
  excludeIds: string[] = [],
  maxDuration?: number
): Stretch {
  let pool = getStretchesForAreas(areas).filter(s => !excludeIds.includes(s.id));
  if (maxDuration) {
    const durationFiltered = pool.filter(s => s.durationSeconds <= maxDuration);
    if (durationFiltered.length > 0) pool = durationFiltered;
  }
  if (pool.length === 0) pool = getStretchesForAreas(areas); // fallback
  return pool[Math.floor(Math.random() * pool.length)];
}

function TimerRing({ progress }: { progress: number }) {
  const R = 108;
  const CIRC = 2 * Math.PI * R;

      duration: 500,
      easing: Easing.linear,
    });

    strokeDashoffset: CIRC * (1 - animProg.value),
  }));

  const AnimatedCircle = Animated.createAnimatedComponent(Circle);

  return (
    <Svg
      width={240}
      height={240}
      style={{ position: 'absolute' }}
    >
      <Circle
        cx={120} cy={120} r={R}
        fill="none"

      />

        cx={120} cy={120} r={R}
        fill="none"
        stroke={Colors.primary}
        strokeWidth={5}
        strokeLinecap="round"
        strokeDasharray={CIRC}
        animatedProps={animatedProps}
        transform="rotate(-90 120 120)"
      />
    </Svg>
  );
}

## FIX 12 — progress.tsx: Fix heatmap cell sizing for small screens

FILE: artifacts/mobile/app/(tabs)/progress.tsx

PROBLEM: 35 cells at fixed 32px overflow on narrow screens like iPhone SE.

FIX STEP 1: Add this import at the top:

FIX STEP 2: Replace the MonthHeatmap component with:

function MonthHeatmap({ sessions }: { sessions: StretchSession[] }) {
  const { width } = useWindowDimensions();
  const COLS = 7;
  const CARD_PADDING = 36;
  const GAP = 4;
  const cellSize = Math.floor((width - CARD_PADDING * 2 - GAP * (COLS - 1)) / COLS);

  const today = new Date();
  const cells = Array.from({ length: 35 }, (_, i) => {
    const d = new Date(today);
    d.setDate(today.getDate() - (34 - i));
    const key = d.toISOString().split('T')[0];
    const count = sessions.filter(s => s.completedAt.startsWith(key)).length;
    const isToday = i === 34;
    return { count, isToday };
  });

  return (
    <View style={{ flexDirection: 'row', flexWrap: 'wrap', gap: GAP }}>
      {cells.map((c, i) => (
        <View
          key={i}
          style={[
            {
              width: cellSize,
              height: cellSize,
              borderRadius: Math.floor(cellSize * 0.2),
            },
            c.count === 0 && { backgroundColor: Colors.bgSurface },
            c.count === 1 && { backgroundColor: 'rgba(58, 122, 92, 0.3)' },
            c.count === 2 && { backgroundColor: 'rgba(58, 122, 92, 0.6)' },
            c.count >= 3 && { backgroundColor: Colors.primary },
            c.isToday && { borderWidth: 2, borderColor: Colors.accent },
          ]}
        />
      ))}
    </View>
  );
}
